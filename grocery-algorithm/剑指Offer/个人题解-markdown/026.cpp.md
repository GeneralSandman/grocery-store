#include <iostream>
#include <vector>
#include <map>
#include <queue>
#include <stack>

using namespace std;

class Node {
public:
    int val;
    Node* next;

    Node() {}

    Node(int _val) {
        val = _val;
        next = NULL;
    }

    Node(int _val, Node* _next) {
        val = _val;
        next = _next;
    }
};

class Solution {
public:
    Node* insert(Node* head, int insertVal) {
        if(head == nullptr) {
            Node* node = new Node(insertVal);
            node->next = node;
            return node;
        }

        Node* pre = head;

        while(pre->next != head) {

            Node* left = pre;
            Node* right = pre->next;

            if(left->val > right->val) {
                if(left->val < insertVal) {
                    break;
                }
                if(right->val > insertVal) {
                    break;
                }
            }

            if(left->val <= insertVal && insertVal <= right->val) {
                break;
            }

            pre = pre->next;

        }


        Node* node = new Node(insertVal, pre->next);
        pre->next = node;

        return head;
    }
};
